## 🏗️ 一、系统架构概览

### 1.1 架构设计理念
- **微服务架构**: 采用 FastAPI 构建高性能 REST API
- **异步处理**: 使用 async/await 提高并发性能
- **数据分离**: PostgreSQL 存储关系数据，Redis 缓存热点数据
- **AI 集成**: 集成 GLM-4.5-Air 大模型提供智能对话能力
- **实时通信**: WebSocket 支持流式对话

### 1.2 技术栈
- **Web 框架**: FastAPI 0.104+
- **ASGI 服务器**: Uvicorn
- **数据库**: PostgreSQL 14+
- **ORM**: SQLAlchemy 2.0
- **缓存**: Redis 7+
- **认证**: JWT
- **AI 服务**: GLM-4.5-Air (智谱AI)
- **任务队列**: Celery (异步任务)
- **API 文档**: OpenAPI/Swagger

```python
# app/websocket/chat.py
from fastapi import WebSocket, WebSocketDisconnect
import json
from typing import List
from app.services.chat import ChatService

class ConnectionManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)

    def disconnect(self, websocket: WebSocket):
        self.active_connections.remove(websocket)

    async def send_personal_message(self, message: str, websocket: WebSocket):
        await websocket.send_text(message)

    async def broadcast(self, message: str):
        for connection in self.active_connections:
            await connection.send_text(message)

manager = ConnectionManager()

@router.websocket("/ws/chat/{conversation_id}")
async def websocket_chat(websocket: WebSocket, conversation_id: str):
    await manager.connect(websocket)
    chat_service = ChatService()
    
    try:
        while True:
            data = await websocket.receive_text()
            message_data = json.loads(data)
            
            # 处理消息
            response = await chat_service.handle_message(
                conversation_id=conversation_id,
                message=message_data
            )
            
            # 发送回复
            await manager.send_personal_message(
                json.dumps(response),
                websocket
            )
            
            # 广播给其他连接（如果是群聊）
            # await manager.broadcast(json.dumps(response))
            
    except WebSocketDisconnect:
        manager.disconnect(websocket)
```

### 4.1 GLM 模型集成

#### 4.1.1 模型服务封装
```python
# app/services/ai/glm_service.py
from zhipuai import ZhipuAI
from typing import List, Dict, Optional, AsyncGenerator
import json
from app.config import settings

class GLMService:
    def __init__(self):
        self.client = ZhipuAI(api_key=settings.ZHIPU_API_KEY)
        self.model = settings.ZHIPU_MODEL
        self.vision_model = settings.ZHIPU_VISION_MODEL
        self.temperature = 0.72
        self.max_tokens = 512
        self.top_p = 0.9

    def _build_messages(
        self,
        system_prompt: str,
        history: List[Dict],
        current_message: str,
        images: Optional[List[str]] = None
    ) -> List[Dict]:
        """构建消息列表"""
        messages = [{"role": "system", "content": system_prompt}]
        
        # 添加历史消息
        for msg in history[-12:]:  # 限制历史长度
            messages.append(msg)
        
        # 添加当前消息（支持多模态）
        if images:
            content = [
                {"type": "text", "text": current_message}
            ]
            for img in images:
                content.append({
                    "type": "image_url",
                    "image_url": {"url": img}
                })
            messages.append({"role": "user", "content": content})
        else:
            messages.append({"role": "user", "content": current_message})
        
        return messages

    async def generate_response(
        self,
        system_prompt: str,
        history: List[Dict],
        current_message: str,
        images: Optional[List[str]] = None,
        stream: bool = False
    ) -> AsyncGenerator[str, None]:
        """生成AI响应"""
        model = self.vision_model if images else self.model
        messages = self._build_messages(
            system_prompt, history, current_message, images
        )
        
        if stream:
            # 流式响应
            response = self.client.chat.completions.create(
                model=model,
                messages=messages,
                temperature=self.temperature,
                max_tokens=self.max_tokens,
                top_p=self.top_p,
                stream=True
            )
            
            for chunk in response:
                delta = chunk.choices[0].delta.content
                if delta:
                    yield delta
                    
        else:
            # 非流式响应
            response = self.client.chat.completions.create(
                model=model,
                messages=messages,
                temperature=self.temperature,
                max_tokens=self.max_tokens,
                top_p=self.top_p
            )
            
            content = response.choices[0].message.content
            if content:
                yield content
```

```python
import redis
import json
from typing import Any, Optional
from app.config import settings

class RedisCache:
    def __init__(self):
        self.redis = redis.Redis.from_url(
            settings.REDIS_URL,
            decode_responses=True
        )
        self.default_ttl = 3600  # 1小时

    async def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        try:
            value = self.redis.get(key)
            if value:
                return json.loads(value)
            return None
        except Exception as e:
            print(f"Redis get error: {e}")
            return None

    async def set(
        self, 
        key: str, 
        value: Any, 
        ttl: Optional[int] = None
    ) -> bool:
        """设置缓存"""
        try:
            json_value = json.dumps(value, ensure_ascii=False)
            ttl = ttl or self.default_ttl
            return self.redis.setex(key, ttl, json_value)
        except Exception as e:
            print(f"Redis set error: {e}")
            return False

    async def delete(self, key: str) -> bool:
        """删除缓存"""
        try:
            return bool(self.redis.delete(key))
        except Exception as e:
            print(f"Redis delete error: {e}")
            return False

    async def invalidate_pattern(self, pattern: str) -> int:
        """批量删除匹配的缓存"""
        try:
            keys = self.redis.keys(pattern)
            if keys:
                return self.redis.delete(*keys)
            return 0
        except Exception as e:
            print(f"Redis invalidate pattern error: {e}")
            return 0

# 缓存装饰器
def cache_response(ttl: int = 3600):
    def decorator(func):
        async def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = f"{func.__name__}:{hash(str(args) + str(kwargs))}"
            
            # 尝试从缓存获取
            cache = RedisCache()
            cached_data = await cache.get(cache_key)
            if cached_data:
                return cached_data
            
            # 执行函数并缓存结果
            result = await func(*args, **kwargs)
            await cache.set(cache_key, result, ttl)
            return result
        
        return wrapper
    return decorator
```

```python
from cryptography.fernet import Fernet
import hashlib
import os

class DataEncryption:
    def __init__(self):
        self.key = Fernet.generate_key()
        self.cipher_suite = Fernet(self.key)
    
    def encrypt(self, data: str) -> str:
        """加密数据"""
        return self.cipher_suite.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted_data: str) -> str:
        """解密数据"""
        return self.cipher_suite.decrypt(encrypted_data.encode()).decode()

class PasswordHasher:
    @staticmethod
    def hash_password(password: str) -> str:
        """密码哈希"""
        salt = os.urandom(32)
        key = hashlib.pbkdf2_hmac('sha256', password.encode(), salt, 100000)
        return salt.hex() + key.hex()
    
    @staticmethod
    def verify_password(password: str, hashed_password: str) -> bool:
        """验证密码"""
        salt = bytes.fromhex(hashed_password[:64])
        key = hashlib.pbkdf2_hmac('sha256', password.encode(), salt, 100000)
        return hashed_password[64:] == key.hex()
```

```python
# app/utils/data_mask.py
import re
from typing import List

class DataMasker:
    @staticmethod
    def mask_phone(phone: str) -> str:
        """手机号脱敏"""
        if len(phone) == 11:
            return phone[:3] + '****' + phone[7:]
        return phone
    
    @staticmethod
    def mask_email(email: str) -> str:
        """邮箱脱敏"""
        if '@' in email:
            name, domain = email.split('@', 1)
            return name[:2] + '*' * (len(name) - 2) + '@' + domain
        return email
    
    @staticmethod
    def mask_content(content: str) -> str:
        """内容脱敏"""
        # 脱敏手机号
        content = re.sub(r'1[3-9]\d{9}', lambda m: DataMasker.mask_phone(m.group()), content)
        # 脱敏邮箱
        content = re.sub(r'[\w\.-]+@[\w\.-]+\.\w+', lambda m: DataMasker.mask_email(m.group()), content)
        return content
```
