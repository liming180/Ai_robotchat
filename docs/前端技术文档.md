## 📱 一、应用概览

### 1.1 应用定位
AI 伴侣应用是一个专注于情感陪伴的聊天应用，采用现代化的前后端分离架构，提供温馨浪漫的 AI 陪伴体验。应用支持多个 AI 伴侣角色、实时聊天、个性化设置等功能。

### 1.2 核心功能
- ✅ 多个预设 AI 伴侣（小暖、小智、小乐等）
- ✅ 自定义 AI 伴侣创建
- ✅ 实时流式聊天
- ✅ 聊天历史记录
- ✅ 亲密度系统
- ✅ 情感感知与回应（新增）
- ✅ 个性化记忆系统（新增）
- ✅ 多模态互动（新增）
- ✅ 场景化陪伴（新增）

### 1.3 技术栈
- **框架**: Vue 3 + Composition API
- **语言**: TypeScript
- **构建工具**: Vite 5
- **路由**: Vue Router 4
- **状态管理**: Pinia
- **UI 框架**: Element Plus + Tailwind CSS
- **HTTP 客户端**: Axios
- **动画**: @vueuse/motion

**useEmotionAnalysis.ts 组合式函数**
```typescript
import { ref, computed } from 'vue'
import { analyzeEmotionApi, recordEmotionApi } from '@/lib/api/emotion'
import type { EmotionType, EmotionRecord } from '@/types'

export function useEmotionAnalysis() {
  const currentEmotion = ref<EmotionType>('neutral')
  const emotionSuggestions = ref<string[]>([])
  const emotionHistory = ref<EmotionRecord[]>([])
  
  // 分析文本情绪
  const analyzeEmotion = async (text: string) => {
    try {
      const result = await analyzeEmotionApi(text)
      currentEmotion.value = result.emotion
      emotionSuggestions.value = result.suggestions
      
      // 记录情绪到历史
      await recordEmotion({
        emotion: result.emotion,
        intensity: result.intensity,
        content: text,
        timestamp: new Date()
      })
    } catch (error) {
      console.error('Emotion analysis failed:', error)
    }
  }
  
  // 获取情绪建议
  const getEmotionSuggestions = (emotion: EmotionType) => {
    // 根据不同情绪返回相应的建议
    const suggestions = {
      happy: ['继续分享你的快乐吧！', '有什么开心的事情吗？'],
      sad: ['我理解你的感受', '需要聊些什么来缓解心情吗？'],
      angry: ['深呼吸，冷静一下', '有什么让你不开心的事情？'],
      anxious: ['别担心，一切都会好起来的', '想聊聊让你焦虑的事情吗？'],
      neutral: ['今天过得怎么样？', '有什么想分享的吗？']
    }
    return suggestions[emotion] || []
  }
  
  // 获取情绪日历数据
  const getEmotionCalendarData = async (dateRange: { start: Date, end: Date }) => {
    try {
      const response = await getEmotionCalendarApi(dateRange)
      emotionHistory.value = response.data
      return response.data
    } catch (error) {
      console.error('Failed to fetch emotion calendar:', error)
      return []
    }
  }
  
  return {
    currentEmotion,
    emotionSuggestions,
    emotionHistory,
    analyzeEmotion,
    getEmotionSuggestions,
    getEmotionCalendarData
  }
}
```

**useMemorySystem.ts 组合式函数**
```typescript
import { ref, computed } from 'vue'
import { memoryApi } from '@/lib/api/memory'
import type { Memory, MemoryCategory } from '@/types'

export function useMemorySystem() {
  const memories = ref<Memory[]>([])
  const selectedCategory = ref<string>('all')
  const newMemory = ref<Partial<Memory>>({})
  
  const memoryCategories: MemoryCategory[] = [
    { id: 'preference', name: '偏好', icon: '❤️' },
    { id: 'habit', name: '习惯', icon: '📝' },
    { id: 'important_date', name: '重要日期', icon: '📅' },
    { id: 'milestone', name: '里程碑', icon: '🏆' }
  ]
  
  // 获取所有记忆
  const fetchMemories = async (userId: string) => {
    try {
      const response = await memoryApi.getMemories(userId)
      memories.value = response.data
    } catch (error) {
      console.error('Failed to fetch memories:', error)
    }
  }
  
  // 添加记忆
  const addMemory = async (memory: Omit<Memory, 'id' | 'createdAt'>) => {
    try {
      const response = await memoryApi.addMemory(memory)
      memories.value.push(response.data)
      return response.data
    } catch (error) {
      console.error('Failed to add memory:', error)
      throw error
    }
  }
  
  // 更新记忆
  const updateMemory = async (id: string, updates: Partial<Memory>) => {
    try {
      const response = await memoryApi.updateMemory(id, updates)
      const index = memories.value.findIndex(m => m.id === id)
      if (index !== -1) {
        memories.value[index] = response.data
      }
    } catch (error) {
      console.error('Failed to update memory:', error)
      throw error
    }
  }
  
  // 删除记忆
  const deleteMemory = async (id: string) => {
    try {
      await memoryApi.deleteMemory(id)
      memories.value = memories.value.filter(m => m.id !== id)
    } catch (error) {
      console.error('Failed to delete memory:', error)
      throw error
    }
  }
  
  // 根据关键词检索记忆
  const searchMemories = (query: string) => {
    const lowerQuery = query.toLowerCase()
    return memories.value.filter(memory =>
      memory.content.toLowerCase().includes(lowerQuery) ||
      memory.type.toLowerCase().includes(lowerQuery)
    )
  }
  
  // 获取AI个性化提示
  const getPersonalizedPrompt = async (companionId: string) => {
    try {
      const response = await memoryApi.getPersonalizedPrompt(companionId)
      return response.data
    } catch (error) {
      console.error('Failed to get personalized prompt:', error)
      return ''
    }
  }
  
  const filteredMemories = computed(() => {
    if (selectedCategory.value === 'all') {
      return memories.value
    }
    return memories.value.filter(m => m.type === selectedCategory.value)
  })
  
  return {
    memories,
    selectedCategory,
    newMemory,
    memoryCategories,
    filteredMemories,
    fetchMemories,
    addMemory,
    updateMemory,
    deleteMemory,
    searchMemories,
    getPersonalizedPrompt
  }
}
```

**useSceneService.ts 组合式函数**
```typescript
import { ref, onMounted } from 'vue'
import { sceneApi } from '@/lib/api/scene'

export function useSceneService() {
  const activeScenes = ref<Scene[]>([])
  const sceneSettings = ref<SceneSettings>({})
  
  // 初始化场景服务
  const initSceneService = async () => {
    try {
      const settings = await sceneApi.getSettings()
      sceneSettings.value = settings
      
      // 监听系统事件
      setupEventListeners()
    } catch (error) {
      console.error('Failed to init scene service:', error)
    }
  }
  
  // 设置场景
  const setScene = async (sceneType: SceneType, options?: SceneOptions) => {
    try {
      const response = await sceneApi.setScene(sceneType, options)
      activeScenes.value = response.data
      return response.data
    } catch (error) {
      console.error('Failed to set scene:', error)
      throw error
    }
  }
  
  // 获取场景推荐
  const getSceneRecommendations = async (time: Date, userState: UserState) => {
    try {
      const response = await sceneApi.getRecommendations({
        time,
        userState,
        currentScenes: activeScenes.value
      })
      return response.data
    } catch (error) {
      console.error('Failed to get scene recommendations:', error)
      return []
    }
  }
  
  const setupEventListeners = () => {
    // 时间变化监听
    setInterval(() => {
      const now = new Date()
      checkSceneTriggers(now)
    }, 60000) // 每分钟检查一次
    
    // 用户活动监听
    window.addEventListener('visibilitychange', handleVisibilityChange)
  }
  
  const checkSceneTriggers = (currentTime: Date) => {
    // 根据时间触发相应场景
    if (currentTime.getHours() >= 22 && currentTime.getHours() < 6) {
      setScene('bedtime_story')
    } else if (currentTime.getHours() >= 6 && currentTime.getHours() < 9) {
      setScene('morning_greeting')
    }
  }
  
  return {
    activeScenes,
    sceneSettings,
    initSceneService,
    setScene,
    getSceneRecommendations
  }
}
```
